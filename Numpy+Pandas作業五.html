import matplotlib.pyplot as plt    # 讀入 Matplotlib 中的 pyplot 函式庫
import numpy as np     # 讀入 Numpy 函式庫
import pandas as pd    # 讀入 Pandas 函式庫
In [ ]:
#======= 作圖設定 ========#
plt.figure(figsize=(10,20))   # 設定圖形大小
labels = ["Personal department","Accounting department",  "Design department"
          ,"Markeiting department","Purchasing department"]   # 指定部門
ratings_A = [8, 5, 10, 12, 5]   # A公司人數
ratings_B = [10, 8, 20, 18, 12]  # B公司人數
colors = ['violet', 'yellow', 'skyblue','lightcoral','lightgreen']  # 圖形顏色
explode = (0, 0, 0, 0.1, 0)  # 將 行銷部門 突出

#======= 開始繪圖 ========#
plt.subplot(2, 1, 1)  # 畫 A公司的圖
plt.title("Company A",fontsize = 20)   # 設定標題名稱及大小
plt.pie(ratings_A, labels = labels,colors=colors,explode=explode,
        autopct='%1.1f%%' ,textprops = {"fontsize" : 20},startangle=90) # 畫圓餅圖

plt.subplot(2, 1, 2)  # 畫 B公司的圖
plt.title("Company B",fontsize = 20)   # 設定標題名稱及大小
plt.pie(ratings_B, labels = labels,colors=colors ,explode=explode,
        autopct='%1.1f%%',textprops = {"fontsize" : 20},startangle=90)  # 畫圓餅圖

plt.axis("equal")  # 設定圓餅圖長寬等長
plt.show()  # 作圖結果
